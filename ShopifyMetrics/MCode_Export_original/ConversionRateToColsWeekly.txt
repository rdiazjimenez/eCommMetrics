let
    // 1. CARGA DE ORIGEN
    RawData = ConversionRateByReferralWeekly,

    // 2. DESPIVOTAR MÉTRICAS
    // Mantenemos fijas las dimensiones de tiempo, fuente y auditoría
    UnpivotedMetrics = Table.UnpivotOtherColumns(RawData, {"week", "referrer_source", "refresh_date_ConversionRate"}, "MetricName", "MetricValue"),

    // 3. CREAR LLAVE DE ATRIBUTO
    // Combinamos la fuente con el nombre de la métrica (ej. Google_sessions)
    MergedReferrerMetric = Table.CombineColumns(UnpivotedMetrics, {"referrer_source", "MetricName"}, Combiner.CombineTextByDelimiter("_", QuoteStyle.None), "ReferrerMetricKey"),

    // 4. REPIVOTAR A COLUMNAS FINALES
    // Transponemos las llaves combinadas para obtener la matriz final por día
    PivotedResults = Table.Pivot(MergedReferrerMetric, List.Distinct(MergedReferrerMetric[ReferrerMetricKey]), "ReferrerMetricKey", "MetricValue", List.Sum)
in
    PivotedResults
