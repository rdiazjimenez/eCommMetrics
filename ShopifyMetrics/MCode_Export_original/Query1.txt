let
    // 1. CONFIGURACIÆN
    
    // 2. QUERY DEFINIDO
    ShopifyQL_Query = "   FROM orders
    SHOW order_count, order.metafields.custom.origen_venta
    SINCE 2026-01-01
    UNTIL 2026-01-31
    LIMIT 50",
    
 
    // 3. ESTRUCTURA DE LA PETICIÆN
    ApiUrl = "https://" & ParamShopName & ".myshopify.com/admin/api/2025-10/graphql.json",
    GraphQL_Query = "{ shopifyqlQuery(query: """ & ShopifyQL_Query & """) { tableData { columns { name displayName dataType } rows } parseErrors } }",
    Payload = Json.FromValue([query = GraphQL_Query]),

    // 4. FUNCIÆN INTERNA DE EJECUCIÆN CON MANEJO DE RATE LIMIT (429)
    ExecuteRequest = (p) =>
        let
            Response = Web.Contents(ApiUrl, [
                Headers = [#"X-Shopify-Access-Token" = ParamAccessToken, #"Content-Type" = "application/json"],
                Content = p,
                ManualStatusHandling = {429}
            ]),
            Meta = Value.Metadata(Response),
            Status = Meta[Response.Status],
            WaitTime = if Status = 429 then Number.From(Record.FieldOrDefault(Meta[Headers], "Retry-After", 2)) else 0,
            Result = if Status = 429 
                     then @Function.InvokeAfter(() => @ExecuteRequest(p), #duration(0, 0, 0, WaitTime))
                     else Json.Document(Response)
        in
            Result,

    // 5. OBTENCIÆN Y RETORNO
    RawResponse = ExecuteRequest(Payload),
    // 5. OBTENCIÆN Y RETORNO
    data = RawResponse[data],
    // 5. OBTENCIÆN Y RETORNO
    shopifyqlQuery = data[shopifyqlQuery],
    // 5. OBTENCIÆN Y RETORNO
    parseErrors = shopifyqlQuery[parseErrors]
in
    parseErrors
