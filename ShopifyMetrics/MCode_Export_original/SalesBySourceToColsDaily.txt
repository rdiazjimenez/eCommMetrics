let
    // 1. CARGA DE ORIGEN
    RawData = SalesBySourceDaily,

    // 2. DESPIVOTAR MÉTRICAS
    // Transformamos las columnas de métricas en filas para facilitar la combinación
    UnpivotedMetrics = Table.UnpivotOtherColumns(RawData, {"day", "channel", "refresh_date_SalesBySource"}, "MetricName", "MetricValue"),

    // 3. CREAR LLAVE DE ATRIBUTO
    // Combinamos el canal con el nombre de la métrica (ej. Online Store_net_sales)
    MergedChannelMetric = Table.CombineColumns(UnpivotedMetrics, {"channel", "MetricName"}, Combiner.CombineTextByDelimiter("_", QuoteStyle.None), "ChannelMetricKey"),

    // 4. REPIVOTAR A COLUMNAS FINALES
    // Expandimos la llave combinada para tener una columna por cada métrica de cada canal
    PivotedResults = Table.Pivot(MergedChannelMetric, List.Distinct(MergedChannelMetric[ChannelMetricKey]), "ChannelMetricKey", "MetricValue", List.Sum)
in
    PivotedResults
